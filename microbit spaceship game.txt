def getStepsUntilNextRefresh():

                    isOn = 5
                display.set_pixel(col-2, row, isOn)

def is_On():
    ret = True
    for i in range(0, asteroidDensity):
        rnd = randint(0, 1)
        if (rnd == 0):
            ret = False
            break
    return ret


def createRow():
    cols = []
    for i in range(0, 9):
        isOn = is_On()
        cols.append(isOn)
    return cols
